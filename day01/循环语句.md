## 循环语句

### if语句

```js
var score=99;
if(score<60){
console.log('不及格');
}
if(score>90){
console.log('优秀');
}


function calcArea(r){
if(!r) return 0;
return r*r*3.14;
}
var num1=calcArea(3);
console.log(num1);
```


### if-else语句
```js
 var score=88;
        if(score<60){
            console.log('不及格');
        }else if(score<80){
            console.log('良好');
        }else if(score<90){
            console.log('优秀');
        }else if(score<=100){
            console.log('太强了');
        }

//判断奇偶 ==相等运算符:只判断值是否相等 ===全等运算符:值和类型都要相同
        var num=19;
        if(num%2===0){
            console.log(num+'是偶数');
        }else{
            console.log(num+'是一个基数');
        }

//计算成人体重 
        //男：身高（cm）-105 = 标准体重（kg）
        //女：身高（cm）-100 = 标准体重（kg）
        var sex='男';
        var height=178;
        var offset=0;
        if(sex==='男'){
            offset=105;
        }else{
            offset=100;
        }
        var weight=height-offset;
        console.log('身高为'+height+'cm的'+sex+'性标准体重为'+weight+'KG');
```
